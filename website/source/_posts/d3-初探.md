## 属性的扩展
```
// 待扩展对象
var configs = {
    a: 1,
    b: 'string',
    c: {
      a: 1,
      b: 2,
      c: {
        a: 1,
        b: 'sss'
      }
    }
  }
// 扩展对象
var options = {
    a: 2,
    c: {
      a: 11111
    }
  }
// 扩展后对象
var merged = {
  a: 2,
    b: 'string',
    c: {
      a: 11111,
      b: 2,
      c: {
        a: 1,
        b: 'sss'
      }
    }
}
```
